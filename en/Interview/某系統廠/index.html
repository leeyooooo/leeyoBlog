    import { Button } from "antd";

    /**
     * 請將下列程式碼轉換為 Functional component 搭配 Hooks
     */
    class Test extends Component {
      constructor() {
        super();
        this.state = {
          data: []
        };
        this.handleClick = this.handleClick.bind(this);
      }

      componentDidMount() {
        this.setState({
          data: [
            { index: 1, label: "AAAA" },
            { index: 2, label: "BBBB" },
            { index: 3, label: "CCCC" }
          ]
        });
      }

      handleClick() {
        this.setState({
          data: [
            { index: 4, label: "DDDD" },
            { index: 5, label: "EEEE" },
            { index: 6, label: "FFFF" }
          ]
        });
      }

      render() {
        const { data } = this.state;
        return (
          <div className="test">
            {data.map((item) => {
              const { index, label } = item;
              return (
                <div className="test__item" key={index}>{`${index}、${label}`}</div>
              );
            })}
            <Button type="primary" onClick={this.handleClick}>
              Change List
            </Button>
          </div>
        );
      }
    }
